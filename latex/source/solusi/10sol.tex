\begin{solution}
Ketika balok belum bergerak, berlaku
\begin{eqnarray*}
F&=&\mu_{s}mg \\
\mu_{s}&=&\frac{F}{mg}\\
&=&\frac{75.0}{(25.0)(9.8)}=0.306
\end{eqnarray*}

Ketika balok sudah bergerak, berlaku
\begin{eqnarray*}
F&=&\mu_{k}mg \\
\mu_{k}&=&\frac{F}{mg}\\
&=&\frac{60.0}{(25.0)(9.8)}=0.245
\end{eqnarray*}
\end{solution}
